  *movie_id

  *title
  *director_id <<FK>>
  description
  duration
  datePublished
}

entity "Movies_Actors" as emoviesactors {
  *movie_id <<FK>>
  *actor_id <<FK>>
}

entity "Directors" as edirectors {
  *director_id
  --
  *name
  birthDate
}

entity "Actors" as eactors {
  *actor_id
  --
  *name
  birthDate
}
